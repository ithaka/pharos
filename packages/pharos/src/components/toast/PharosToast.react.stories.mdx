# Toast

<Canvas withToolbar>
  <Story name="Base">
    <PharosButton
      id="success-toast-button"
      onClick={() => {
        const event = new CustomEvent('pharos-toast-open', {
          detail: {
            content:
              'The item has moved to your <PharosLink href="#" onBackground bold>Workspace</PharosLink>.',
          },
        });
        document.dispatchEvent(event);
      }}
    >
      Have a toast!
    </PharosButton>
