import { Story, Canvas, Meta, ArgsTable } from '@storybook/addon-docs/blocks';

import '../link/pharos-link';

<Meta title="Components/Grid" />

export const gridItems = (rows) => {
  const arr = [];
  rows.forEach((items) => {
    for (let i = 0; i < items; i++) {
      arr.push({ span: 12 / items });
    }
  });
  return arr.map(
    (item) => html`<div
      style="grid-column: span ${item.span}; background: var(--pharos-color-living-coral-90); height: 3rem"
    ></div>`
  );
};
