<Meta title="Components/Grid" />

export const gridItems = (rows) => {
  const arr = [];
  rows.forEach((items) => {
    for (let i = 0; i < items; i++) {
      arr.push({ span: 12 / items, cols: items });
    }
  });
  return arr.map(
    (item) => html`<div class="grid-example__item" style="grid-column: span ${item.span}">
      ${item.cols}
    </div>`
  );
};

export const comboItems = (rows) => {
  const arr = [];
  rows.forEach((item) => {
    arr.push({ span: item, cols: item });
    arr.push({ span: 12 - item, cols: 12 - item });
  });
  return arr.map(
    (item) => html`<div class="grid-example__item" style="grid-column: span ${item.span}">
      Span ${item.cols}
    </div>`
  );
};
